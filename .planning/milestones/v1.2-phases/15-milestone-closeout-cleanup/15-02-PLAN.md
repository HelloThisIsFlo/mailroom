class TriageCategory(BaseModel):
    name: str
    label: str | None = None
    contact_group: str | None = None
    destination_mailbox: str | None = None
    contact_type: Literal["company", "person"] = "company"
    parent: str | None = None
    add_to_inbox: bool = False

class MailroomSectionSettings(BaseModel):
    label_error: str = "@MailroomError"
    label_warning: str = "@MailroomWarning"
    warnings_enabled: bool = True
    provenance_group: str = "Mailroom"

class PollingSettings(BaseModel):
    interval: int = 60
    debounce_seconds: int = 3

class LoggingSettings(BaseModel):
    level: str = "info"

class MailroomSettings(BaseSettings):
    jmap_token: str
    carddav_username: str = ""
    carddav_password: str = ""
    polling: PollingSettings
    triage: TriageSettings
    mailroom: MailroomSectionSettings
    logging: LoggingSettings
